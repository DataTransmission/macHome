# follow the stdout/stderr of a running batch job
$ bpeek -f <JobID> | grep <keyword>

$ bkill <job#>

$ bfill

#check current owner job (customized)
$ bjobs -o "jobid project queue stat submit_time mem delimiter=','" -u $USER

# list the nodes
$ bhosts -l
