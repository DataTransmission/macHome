# ignore case 
$ grep -i
 
# exclude directory including pattern 'word'
$ grep --exclude-dir=*word* 'some pattern' /path/to/search

# exclude the lines with pattern 'exlude' but include 'include' in
# current directory
$ grep 'include' * | grep -v 'exclude'

# grep multiple words under <dir>
$ grep 'word1' <dir> | grep 'word2' | grep 'word3'

# grep 5 lines before and 10 lines after a 'word'
$ grep -B 5 -A 10 'word'

# color grep with linenumber
$ grep --color=always -n -r
